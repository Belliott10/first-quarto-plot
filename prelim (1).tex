% Custom proof environment with explicit reference to theorem/corollary/etc.
\makeatletter
\newenvironment{customproof}[1]{%
  \par\pushQED{\qed}%
  \normalfont \topsep6\p@\@plus6\p@\relax
  \trivlist
  \item[\hskip\labelsep
        \bfseries Proof of #1.]%
}{%
  \popQED\endtrivlist\@endpefalse
}
